36 """

43 Returns: area (Tensor[N]): area for each box

48def calculate_rps_iou(rps_0: np.ndarray, rps_1: np.ndarray) -> np.ndarray: 

49 """ 

50 :param rps_0: (n, 4) 

51 :param rps_1 (m, 4) 

52 :returns: (n, m) 

53 Copy from https://www.codeleading.com/article/54902955171/ 

54 """ 

55 lt = np.maximum(rps_0[:, None, :2], rps_1[:, :2]) # left_top (x, y) 

56 rb = np.minimum(rps_0[:, None, 2:], rps_1[:, 2:]) # right_bottom (x, y) 

57 wh = np.maximum(rb - lt + 1, 0) # inter_area (w, h) 

58 inter_areas = wh[:, :, 0] * wh[:, :, 1] # shape: (n, m) 

59 box_areas = (rps_0[:, 2] - rps_0[:, 0] + 1) * (rps_0[:, 3] - rps_0[:, 1] + 1) 

60 gt_areas = (rps_1[:, 2] - rps_1[:, 0] + 1) * (rps_1[:, 3] - rps_1[:, 1] + 1) 

61 iou = inter_areas / (box_areas[:, None] + gt_areas - inter_areas) 

62 return np.round(iou, 3)
